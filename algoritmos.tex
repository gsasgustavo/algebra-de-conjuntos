  \section{Insere}

  \begin{algorithm}[H]
    \label{alg1}
    \caption {\textsc{Insere($A, a$)}}
    \Entrada {$A$ é um conjunto, $a$ é um valor que será inserido em $A$}
    \Saida{}
    \SetAlgoLined
      \Inicio {
        \Se {$a$ \notin $A$}{
            $N$ \leftarrow $|A|$ \\
            $A_N_+_1$ \leftarrow a \\
        }
      }
      \end{algorithm}
