  \section{Insere}

  \begin{algorithm}[H]
    \label{alg1}
    \caption {\textsc{Insere($A, a$)}}
    \Entrada {$A$ é um conjunto, $a$ é um valor que será inserido em $A$}
    \Saida{}
    \SetAlgoLined
      \Inicio {
        \Se {$a$ \notin $A$}{
            $N$ \leftarrow $|A|$ \\
            $A_N_+_1$ \leftarrow a \\
        }
      }
      \end{algorithm}
      
    \section{União}
    
    \begin{algorithm}[H]
    \caption{\textsc{União}($A$ e $B$)}
    \Entrada{A e B são conjuntos}
    \Saida{O conjunto resultante da União entre os conjuntos A e B}
    \Inicio{
        C \leftarrow \{\}  \\
        \Para{cada $a \epsilon A$}{ 
            Insere($C,a$)
        }
        \Para{cada $b \epsilon B$}{ 
            Insere($C,b$)
        }
        $retorne $C$ $
    }
    \end{algorithm}
